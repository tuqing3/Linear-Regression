    #转换数据
    X_binned = enc.fit_transform(X)
    line_binned = enc.transform(line)

    #全数据集上的交叉验证
    cvresult = CVS(LinearR_,X_binned,y,cv=5)
    score.append(cvresult.mean())
    var.append(cvresult.var())
